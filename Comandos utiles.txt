Para poder ver el codigo en Google Collab:

from pygments import highlight
from pygments.lexers import JavaLexer
from pygments.formatters import HtmlFormatter
from IPython.display import HTML
import glob

style = "monokai"  # podés probar otros: "native", "fruity", "vim", etc.

# Recorre todos los archivos .java y los muestra con resaltado
for file in glob.glob("*.java"):
    code = open(file).read()
    formatter = HtmlFormatter(style=style, full=True, noclasses=True)
    display(HTML(f"<h3>{file}</h3>" + highlight(code, JavaLexer(), formatter)))
